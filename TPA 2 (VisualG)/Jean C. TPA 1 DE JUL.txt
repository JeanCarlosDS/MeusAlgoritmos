1- Algoritmo "Numeros maiores ou iguais"
var
   x, y: inteiro
inicio
   escreva ("Digite os 2 números: ")
   leia(x, y)
   se (x > y) entao
      escreval (x," é maior que",y)
   senao
      escreval (y," é maior que",x)
   fimse
   se (x = y) então
      escreval ("OBS: Os dois são iguais!")
   fimse
fimalgoritmo

4- Algoritmo "Triangulos"
Var
   A, B, C: inteiro
Inicio
   escreval ("Quais os lados do triangulo?")
   leia (A, B, C)
   se ((A < C + B) e (B < C + A) e (C < B + A)) entao
      se ((A = B) e (B = C)) entao
         escreval ("É um triangulo equilátero")
      senao
         se ((A = B) ou (A = C) ou (B = C)) entao
            escreval ("É um triangulo Isoceles")
         senao
            escreval ("É um triangulo Escaleno")
         fimse
      fimse
   senao
      escreval ("Isso não é um triangulo")
   fimse
Fimalgoritmo
